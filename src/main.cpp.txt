#include <Wire.h>
#include <LiquidCrystal_I2C.h>
#include <EEPROM.h>
#include <Keypad.h>

// ── Keypad Setup ──
const byte ROWS = 4, COLS = 4;
char keys[ROWS][COLS] = {
  {'1','2','3','A'},
  {'4','5','6','B'},
  {'7','8','9','C'},
  {'*','0','#','D'}
};
byte rowPins[ROWS] = {13, 12, 14, 27};
byte colPins[COLS] = {26, 25, 33, 32};
Keypad kpd = Keypad(makeKeymap(keys), rowPins, colPins, ROWS, COLS);

// ── LCD ──
LiquidCrystal_I2C lcd(0x27, 16, 2);

// ── EEPROM Layout ──
#define ADDR_BOOT    0
#define ADDR_ULEN    1
#define ADDR_UNAME   2
#define ADDR_PINLEN  17
#define ADDR_PIN     18
#define SLOT_START   26
#define SLOT_SIZE    16
#define MAX_SLOTS    10

const byte KEY = 0xAD;

char uname[16];
byte pinLen;

// ── Buzzer ──
#define BUZZER 15
void beep(int f, int d) { tone(BUZZER, f, d); }

// ── LCD Helper ──
void sc(const char* a, const char* b) {
  lcd.clear();
  lcd.setCursor(0,0); lcd.print(a);
  lcd.setCursor(0,1); lcd.print(b);
}

// ── EEPROM Helpers ──
int sa(int s) { return SLOT_START + s * SLOT_SIZE; }

bool sf(int s) {
  byte l = EEPROM.read(sa(s));
  return l > 0 && l <= 15 && l != 255;
}

void xs(const char* p, int l, int a) {
  for(int i=0;i<l;i++) EEPROM.write(a+i, p[i]^KEY);
}

String xr(int a, int l) {
  String s="";
  for(int i=0;i<l;i++) s+=(char)(EEPROM.read(a+i)^KEY);
  return s;
}

// ── Multi-Tap Input ──
bool isNumberMode = false;  // 🔢 global

String getInput(const char* prompt, bool hide) {
  String input = "";
  lcd.clear();
  lcd.setCursor(0,0); lcd.print(prompt);
  lcd.setCursor(0,1);

  unsigned long lastPressTime = 0;
  char lastKey = 0;
  int tapIndex = 0;

  bool isUpper = false;

  // show default mode
  lcd.setCursor(13,0);
  lcd.print("T"); // Text mode

  while (true) {
    char k = kpd.getKey();
    if (!k) continue;

    unsigned long now = millis();

    // 🔢 Toggle NUMBER MODE
    if (k == 'C') {
      isNumberMode = !isNumberMode;

      lcd.setCursor(13,0);
      lcd.print(isNumberMode ? "N" : "T");

      beep(2000, 100);
      continue;
    }

    // 🔤 Toggle UPPERCASE
    if (k == 'D') {
      isUpper = !isUpper;

      lcd.setCursor(14,0);
      lcd.print(isUpper ? "A" : "a");

      beep(1800, 100);
      continue;
    }

    // 🔢 NUMBER MODE (direct input)
    if (isNumberMode && isdigit(k)) {
      input += k;
      lastKey = 0;

      lcd.setCursor(0,1);
      lcd.print("                ");
      lcd.setCursor(0,1);
      lcd.print(input);

      beep(1500, 80);
      continue;
    }

    String chars = "";

    // 🔤 TEXT MODE mappings
    if (k == '2') chars = "abc";
    else if (k == '3') chars = "def";
    else if (k == '4') chars = "ghi";
    else if (k == '5') chars = "jkl";
    else if (k == '6') chars = "mno";
    else if (k == '7') chars = "pqrs";
    else if (k == '8') chars = "tuv";
    else if (k == '9') chars = "wxyz";
    else if (k == '0') chars = " ";
    else if (k == '1') chars = "1@._";
    else if (k == 'A') chars = "#$%";
    else if (k == 'B') chars = "!?";

    // ⬅️ Backspace
    else if (k == '*') {
      if (input.length() > 0) input.remove(input.length() - 1);
      lastKey = 0;
      continue;
    }

    // ✅ Confirm
    else if (k == '#') {
      if (input.length() > 0) return input;
    }

    else continue;

    // 🔁 Multi-tap logic
    if (k == lastKey && (now - lastPressTime) < 800) {
      tapIndex = (tapIndex + 1) % chars.length();
      input.remove(input.length() - 1);
    } else {
      tapIndex = 0;
    }

    char c = chars[tapIndex];

    // 🔠 Apply uppercase
    if (isUpper && isalpha(c)) {
      c = toupper(c);
    }

    input += c;

    lastKey = k;
    lastPressTime = now;

    // 📺 Display
    lcd.setCursor(0,1);
    lcd.print("                ");
    lcd.setCursor(0,1);

    if (hide) {
      for (int i = 0; i < input.length(); i++) lcd.print("*");
    } else {
      lcd.print(input);
    }

    beep(1000, 40);
  }
}
// ── Auth ──
bool auth() {
  String entered = getInput("Enter PIN:", true);
  if(entered == "") return false;

  String stored = xr(ADDR_PIN, pinLen);

  if(entered == stored) {
    sc("Access Granted!", "");
    beep(2000, 400);
    delay(1500);
    return true;
  }

  sc("Wrong PIN!", "Denied!");
  beep(200, 800);
  delay(2000);
  return false;
}

// ── First Boot ──
void firstBoot() {
  sc("Welcome!", "Setup");
  delay(2000);

  while(true) {
    String n = getInput("Username:", false);
    if(n.length() > 0) {
      EEPROM.write(ADDR_ULEN, n.length());
      xs(n.c_str(), n.length(), ADDR_UNAME);
      strncpy(uname, n.c_str(), 15);
      break;
    }
  }

  while(true) {
    String p = getInput("New PIN:", true);
    if(p.length() >= 4) {
      String p2 = getInput("Confirm PIN:", true);
      if(p == p2) {
        pinLen = p.length();
        EEPROM.write(ADDR_PINLEN, pinLen);
        xs(p.c_str(), pinLen, ADDR_PIN);
        break;
      }
    }
  }

  EEPROM.write(ADDR_BOOT, 1);
  EEPROM.commit();
}

// ── Load ──
void loadAll() {
  int ul = EEPROM.read(ADDR_ULEN);

  if(ul > 0 && ul <= 15 && ul != 255) {
    String u = xr(ADDR_UNAME, ul);
    strncpy(uname, u.c_str(), 15);
  } else strcpy(uname, "User");

  pinLen = EEPROM.read(ADDR_PINLEN);
  if(pinLen > 8 || pinLen == 255) pinLen = 0;
}

// ── View ──
void viewFlow() {
  if(!auth()) return;

  String s = getInput("Slot:", false);
  if(s == "") return;

  int slot = s.toInt();

  if(slot >= 0 && slot < MAX_SLOTS && sf(slot)) {
    int a = sa(slot);
    int l = EEPROM.read(a);
    String d = xr(a+1, l);

    lcd.clear();
    lcd.print("Slot " + String(slot));
    lcd.setCursor(0,1);
    lcd.print(d);
    delay(5000);
  } else {
    sc("Invalid slot!", "");
    delay(2000);
  }
}

// ── Insert ──
void insertFlow() {
  String s = getInput("Slot:", false);
  if(s == "") return;

  int slot = s.toInt();

  if(slot < 0 || slot >= MAX_SLOTS || sf(slot)) {
    sc("Invalid slot!", "");
    delay(2000);
    return;
  }

  String pass = getInput("Password:", false);
  if(pass == "") return;

  int a = sa(slot);

  EEPROM.write(a, pass.length());
  xs(pass.c_str(), pass.length(), a+1);
  EEPROM.commit();

  sc("Saved!", "");
  delay(2000);
}

// ── Delete ──
void deleteFlow() {
  if(!auth()) return;

  String s = getInput("Delete:", false);
  if(s == "") return;

  int slot = s.toInt();

  if(slot >= 0 && slot < MAX_SLOTS && sf(slot)) {
    int a = sa(slot);
    for(int i=0;i<SLOT_SIZE;i++) EEPROM.write(a+i, 0);
    EEPROM.commit();
    sc("Deleted!", "");
    delay(2000);
  } else {
    sc("Invalid!", "");
    delay(2000);
  }
}

// ── Setup ──
void setup() {
  Wire.begin(21,22);
  lcd.init();
  lcd.backlight();
  EEPROM.begin(512);

  pinMode(BUZZER, OUTPUT);

  if(EEPROM.read(ADDR_BOOT) != 1) firstBoot();
  else loadAll();

  lcd.clear();
  lcd.print("Hello ");
  lcd.print(uname);
  lcd.setCursor(0,1);
  lcd.print("Vault Ready");
  delay(2000);
}

// ── Loop ──
void loop() {
  sc("1.View 2.Save", "3.Delete");

  while(true) {
    char k = kpd.getKey();

    if(k == '1') { viewFlow(); return; }
    if(k == '2') { insertFlow(); return; }
    if(k == '3') { deleteFlow(); return; }
  }
}
